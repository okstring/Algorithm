## 괄호 변환



```swift
func solution(_ p:String) -> String {
    if p == "" { return "" }
    var v = p
    var u = one(v: &v)
    
    if isCorrectString(u) {
        return u + solution(v)
    } else {
        var tmpStr = "("
        tmpStr += solution(v) + ")"
        _ = u.removeFirst()
        _ = u.removeLast()
        return tmpStr + u.map({
            return $0 == "(" ? ")" : "("
        }).joined()
    }
}

func one(v: inout String) -> String {
    if v == "" { return "" }
    var o = 0
    var c = 0
    var u = ""
    
    repeat {
        let e = v.removeFirst()
        u += String(e)
        if e == "(" {
            o += 1
        } else {
            c += 1
        }
    } while o != c
    
    return u
}

func isCorrectString(_ str: String) -> Bool {
    if str.isEmpty { return true }
    var tmpArr = Array(str)
    var stack = [String]()
    while !tmpArr.isEmpty {
        let e = tmpArr.removeLast()
        if e == ")" {
            stack.append(String(e))
        } else if e == "(" {
            if stack.isEmpty {
                return false
            }
            _ = stack.popLast()!
        }
    }
    
    return stack.isEmpty ? true : false
}
```

```swift
func solution(_ p:String) -> String {
    if p.count < 1 {return ""}

    var count = 0, index = p.startIndex
    repeat{
        count += String(p[index]) == "(" ? 1 : -1
        index = p.index(after: index)
    } while count != 0

    var u = String(p[..<index]), v = String(p[index...])
    if String(u.first!) == "("{
        return u + solution(v)
    }else{
        u.removeLast()
        u.removeFirst()
        return "(\(solution(v)))\(u.map{String($0) == "(" ? ")" : "("}.joined())"
    }
}
```
